# =========================
# üßæ Setup Logging
# =========================
$logFile = "C:\Logs\PrivacyConfig.log"
Start-Transcript -Path $logFile
Write-Host "üï∂Ô∏è Starting privacy and update configuration..." -ForegroundColor Cyan

# =========================
# üõ†Ô∏è Helper: Safe Registry Setter
# =========================
function Set-RegistryValueSafe {
    param (
        [string]$Path,
        [string]$Name,
        [Object]$Value,
        [string]$Type = "DWord"
    )
    try {
        Set-ItemProperty -Path $Path -Name $Name -Type $Type -Value $Value
        Write-Output "üîê $Name set to $Value at $Path" | Out-File $logFile -Append
    } catch {
        Write-Warning "‚ö†Ô∏è Failed to set $Name at $Path: $_"
    }
}

# =========================
# üì¶ Disable Consumer Features & Recommendations
# =========================
Set-RegistryValueSafe "HKLM:\SOFTWARE\Policies\Microsoft\Windows\CloudContent" "DisableWindowsConsumerFeatures" 1
Set-RegistryValueSafe "HKCU:\SOFTWARE\Microsoft\Windows\CurrentVersion\ContentDeliveryManager" "SystemPaneSuggestionsEnabled" 0
Set-RegistryValueSafe "HKCU:\SOFTWARE\Microsoft\Windows\CurrentVersion\ContentDeliveryManager" "SubscribedContent-310093Enabled" 0

# =========================
# üîß Disable Scheduled Update Tasks
# =========================
$tasksToDisable = @(
    "\Microsoft\Windows\Setup\EOSNotify",
    "\Microsoft\Windows\Setup\EOSNotify2",
    "\Microsoft\Windows\Setup\OobeUpdater"
)
foreach ($task in $tasksToDisable) {
    try {
        Disable-ScheduledTask -TaskName $task
        Write-Output "üõë Disabled scheduled task: $task" | Out-File $logFile -Append
    } catch {
        Write-Warning "‚ö†Ô∏è Failed to disable task $task: $_"
    }
}

# =========================
# ü™ü Configure Windows Update Behavior
# =========================
# Turn off automatic updates (requires AU key setup)
$wuPath = "HKLM:\SOFTWARE\Policies\Microsoft\Windows\WindowsUpdate\AU"
if (-not (Test-Path $wuPath)) {
    New-Item -Path $wuPath -Force
}
Set-RegistryValueSafe $wuPath "NoAutoUpdate" 1

# Enable Microsoft Update Service Manager
try {
    (New-Object -com "Microsoft.Update.ServiceManager").AddService2("7971f918-a847-4430-9279-4a52d1efe18d", 7, "")
    Write-Output "üåê Microsoft Update Service enabled" | Out-File $logFile -Append
} catch {
    Write-Warning "‚ö†Ô∏è Failed to enable Microsoft Update Service: $_"
}

# =========================
# ‚úÖ Wrap-Up
# =========================
Stop-Transcript
Write-Host "üîí Privacy and update configuration complete." -ForegroundColor Green
